cmake_minimum_required(VERSION 3.5)

include("../CMake_inc.cmake")
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

	GWColor.cpp
	GWMatrix.cpp
	GWTransform.cpp

	GWImage.cpp
	GWSphericalHarmonics.cpp
	GWResource.cpp
	GWModel.cpp
	GWScene.cpp
	${TDMOTION_DIR}/TDMotion.cpp

add_executable(gw_test gw_test.cpp)
